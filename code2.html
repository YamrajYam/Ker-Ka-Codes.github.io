                            from tkinter import *
                            from tkinter import messagebox
                            from turtle import width
                            import math
                            #create window
                            root=Tk()
                            
                            #naming window
                            root.title("Claculator")
                            
                            #getting window icon
                            img=PhotoImage(file="calculator.png")
                            root.iconphoto(False,img) 
                            
                            #window geometry
                            root.configure(height=400 ,width=300,background="#202120")
                            root.minsize(300,400)
                            root.maxsize(300,400)
                            #output frame
                            disp=Frame(root,width=280,height= 100 , background="black",highlightbackground="black", highlightthickness=2)
                            disp.pack()
                            disp.place(x=10,y=10)
                            
                            
                            #dev info
                            dev=Label(root,text="Coded by Kuldeep Kumar",font="arial 10",background="black",foreground="white")
                            dev.pack()
                            dev.place(x=75,y=370)
                            
                            
                            #font for user input
                            font1="arial 16 "
                            
                            #numarioc pad frame
                            pad=Frame(root,width=280,height=250, background="black",highlightbackground="white", highlightthickness=2)
                            pad.pack()
                            pad.place(x=10,y=115)
                            
                            
                            #display user input
                            inp=Entry(disp, font=font1,background="white",foreground="black")
                            inp.pack()
                            inp.place(x=5,y=5,height=85,width=265)
                            
                            
                            inpu=(inp.get())
                            
                            #getting images fro numbur buttons
                            p1=PhotoImage(file="1.png")
                            p2=PhotoImage(file="2.png")
                            p3=PhotoImage(file="3.png")
                            p4=PhotoImage(file="4.png")
                            p5=PhotoImage(file="5.png")
                            p6=PhotoImage(file="6.png")
                            p7=PhotoImage(file="7.png")
                            p8=PhotoImage(file="8.png")
                            p9=PhotoImage(file="9.png")
                            p0=PhotoImage(file="0.png")
                            pc=PhotoImage(file="c.png")
                            pe=PhotoImage(file="e.png")
                            pc=PhotoImage(file="c.png")
                            pmi=PhotoImage(file="mi.png")
                            pmu=PhotoImage(file="mu.png")
                            pd=PhotoImage(file="d.png")
                            pp=PhotoImage(file="p.png")
                            pr=PhotoImage(file="r.png")
                            ps=PhotoImage(file="s.png")
                            
                            #assigning button to images
                            
                            #action for buttons
                            def inse():
                                inp.insert("end","1")
                            
                            def inse2():
                                inp.insert("end","2")
                            
                            def inse3():
                                inp.insert("end","3")
                            
                            def inse4():
                                inp.insert("end","4")
                            
                            def inse5():
                                inp.insert("end","5")
                            
                            def inse6():
                                inp.insert("end","6")
                            
                            def inse7():
                                inp.insert("end","7")
                            
                            def inse8():
                                inp.insert("end","8")
                            
                            def inse9():
                                inp.insert("end","9")
                            
                            def inse0():
                                inp.insert("end","0")
                            
                            def insep():
                                inp.insert("end","+")
                            
                            
                            def insemi():
                                inp.insert("end","-")
                            
                            
                            def insemu():
                                inp.insert("end","*")
                            
                            
                            def insedi():
                                inp.insert("end","/")
                            
                            
                            def inser():
                                r=int(inp.get())
                                r=math.sqrt(r)
                                clear()
                                inp.insert("end",r)
                            
                            
                            def inses():
                                n=int(inp.get())
                                s=n**2
                                clear()
                                inp.insert("end",s)
                            
                            
                            
                            
                            
                            #number button
                            b1=Button(pad,image=p1,command=inse,height=35,width=60)
                            b1.pack()
                            b1.place(x=5,y=150)
                            
                            
                            b2=Button(pad,image=p2,command=inse2,height=35,width=60)
                            b2.pack()
                            b2.place(x=75,y=150)
                            
                            b3=Button(pad,image=p3,command=inse3,height=35,width=60)
                            b3.pack()
                            b3.place(x=145,y=150)
                            
                            b4=Button(pad,image=p4,command=inse4,height=35,width=60)
                            b4.pack()
                            b4.place(x=5,y=100)
                            
                            b5=Button(pad,image=p5,command=inse5,height=35,width=60)
                            b5.pack()
                            b5.place(x=75,y=100)
                            
                            b6=Button(pad,image=p6,command=inse6,height=35,width=60)
                            b6.pack()
                            b6.place(x=145,y=100)
                            
                            b7=Button(pad,image=p7,command=inse7,height=35,width=60)
                            b7.pack()
                            b7.place(x=5,y=50)
                            
                            b8=Button(pad,image=p8,command=inse8,height=35,width=60)
                            b8.pack()
                            b8.place(x=75,y=50)
                            
                            b9=Button(pad,image=p9,command=inse9,height=35,width=60)
                            b9.pack()
                            b9.place(x=145,y=50)
                            
                            b0=Button(pad,image=p0,command=inse0,height=35,width=60)
                            b0.pack()
                            b0.place(x=5,y=200)
                            
                            #function for buttons
                            def clear():
                                inp.delete(0,END)
                            
                            
                            def enter():
                                try:
                                    a=(inp.get())
                                    result=eval(a)
                                    clear()
                                    inp.insert(0,result)
                                except:
                                    messagebox.showerror("Error" ,
                                                         "Invalid Input")
                            
                                
                            
                            
                            
                            
                            
                            #action image
                            br=Button(pad,image=pr,command=inser,height=35,width=60)
                            br.pack()
                            br.place(x=5,y=5)
                            
                            
                            
                            
                            bs=Button(pad,image=ps,command=inses,height=35,width=60)
                            bs.pack()
                            bs.place(x=75,y=5)
                            
                            bd=Button(pad,image=pd,command=insedi,height=35,width=60)
                            bd.pack()
                            bd.place(x=145,y=5)
                            
                            
                            bmi=Button(pad,image=pmi,command=insemi,height=37,width=47)
                            bmi.pack()
                            bmi.place(x=220,y=50)
                            
                            bmu=Button(pad,image=pmu,command=insemu,height=35,width=47)
                            bmu.pack()
                            bmu.place(x=220,y=5)
                            
                            ba=Button(pad,image=pp,command=insep,height=37,width=47)
                            ba.pack()
                            ba.place(x=220,y=100)
                            
                            be=Button(pad,image=pe,command=enter,height=85,width=47)
                            be.pack()
                            be.place(x=220,y=150)
                            
                            
                            bc=Button(pad,image=pc,command=clear,height=35,width=130)
                            bc.pack()
                            bc.place(x=75,y=200)
                            
                            
                            
                            root.mainloop()

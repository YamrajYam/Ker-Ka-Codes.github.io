import mysql.connector as ms
mydb = ms.connect(
    host="localhost",
    user="root",
    password="your password",
    database="nameofdatabase") 

w=mydb.cursor()
#add details
def adddetails(P_ID, NAME,CATAGORY, BRAND,QUANTITY,DATE_OF_IMPORT):
    a="insert into IMPORTED_MATERIAL(P_ID, NAME,CATAGORY, BRAND,QUANTITY,DATE_OF_IMPORT) values(%s,%s,%s,%s,%s,%s)"
    w.execute(a,(P_ID, NAME,CATAGORY, BRAND,QUANTITY,DATE_OF_IMPORT))
    mydb.commit()

#accesing data
def disall():
    b="select * from imported_material"
    w.execute(b)
    rows=w.fetchall()
    return rows



#delete data
def deleted(z):
    c="delete from imported_material where P_ID ={}".format(z)
    w.execute(c)
    mydb.commit()

#delete all data from database
def deleteall():
    d="delete from imported_material"
    w.execute(d)
    mydb.commit()

#display data from P_ID
def disq(x):
    e="select * from imported_material where P_ID={}".format(x)
    w.execute(e)
    f=w.fetchone()
    return f
    #print(f)

#update data of specific P_ID
def modi(P_ID, NAME,CATAGORY, BRAND,QUANTITY,DATE_OF_IMPORT):
    f="update imported_material set NAME=%s,CATAGORY=%s, BRAND=%s,QUANTITY=%s,DATE_OF_IMPORT=%s where P_ID=%s"
    w.execute(f,( NAME,CATAGORY, BRAND,QUANTITY,DATE_OF_IMPORT,P_ID))
    mydb.commit()

#filters for viewing sorted data


def sortidasc():
    w.execute("select * from imported_material order by P_ID ASC")
    rows=w.fetchall()
    return rows


def sortiddesc():
    w.execute("select * from imported_material order by P_ID DESC")
    rows=w.fetchall()
    return rows


def sortqasc():
    w.execute("select * from imported_material order by QUANTITY ASC")
    rows=w.fetchall()
    return rows


def sortqdesc():
    w.execute("select * from imported_material order by QUANTITY DESC")
    rows=w.fetchall()
    return rows
def sortdoiasc():
    w.execute("select * from imported_material order by DATE_OF_IMPORT ASC")
    rows=w.fetchall()
    return rows
def sortdoidesc():
    w.execute("select * from imported_material order by DATE_OF_IMPORT DESC")
    rows=w.fetchall()
    return rows

def addd():
    check=e1.get()
    check2=e3.get()
    if check=="" or e2.get()=="" or e3.get()=="" or e4.get()=="" or catg.get()=="" or cal.get()=="":
        messagebox.showerror("Error","Kindely Enter all values")
        return
    if check.isdigit() and check2.isdigit():
        adddetails(check,(e2.get()).title(),(catg.get()).title(),(e4.get()).title(),(e3.get()).title(),(cal.get()).title())
        messagebox.showinfo("Success", "Record Added")
    else:
        messagebox.showerror("Error","Only Numaric value allowed for P ID and Quantity")
    clearf()
    dispall()
def upd():
    check=e1.get()    
    if check=="":
        messagebox.showerror("""Error", "Kindly enter P ID,of
                             Which details is to be updated""")
        return
    
    if check.isdigit():
        convert=str(check)
        for row in disall():
            if check==row[0]:
                messagebox.showerror("Error", "Record does not exist")
        return
    if check.isdigit():
        convert=str(check)
        modi(convert,(e2.get()).title(),(catg.get()).title(),(e4.get()).title(),(e3.get()).title(),(cal.get()).title())
        messagebox.showinfo("Success", "Record Updated")
        clearf()
        dispall()
